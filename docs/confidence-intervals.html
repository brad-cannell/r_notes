Assigning heights to the generated participants at random (within certain parameters).

set.seed(123)
random_data <- random_data %>% 
  mutate(
    height = if_else(
      sex == "Female",
      rnorm(n, 71.00, 4.32),
      rnorm(n, 64.30, 2.11)
    )
  )
random_data %>% 
  group_by(sex) %>% 
  mean_table(height) %>% 
  select(group_cat, mean, sd, lcl, ucl)
